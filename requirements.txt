from flask import Flask, jsonify
from flask_cors import CORS
import requests
import os
from datetime import datetime

app = Flask(__name__)
CORS(app)

# =========================
# CONFIG
# =========================
NASA_API_KEY = os.getenv("NASA_API_KEY", "DEMO_KEY")

NASA_URL = "https://api.nasa.gov/planetary/earth/imagery"
ELEVATION_URL = "https://api.open-elevation.com/api/v1/lookup"


# =========================
# HOME ROUTE
# =========================
@app.route("/")
def home():
    return jsonify({
        "service": "OmniScan-XR2",
        "status": "running",
        "version": "2.0",
        "features": ["NASA imagery", "Elevation scan", "Geo JSON output"]
    })


# =========================
# TERRAIN CLASSIFIER (simple upgrade)
# =========================
def classify_terrain(elevation):
    if elevation < 10:
        return "coastal/flat"
    elif elevation < 200:
        return "lowland/urban"
    elif elevation < 1000:
        return "hills"
    else:
        return "mountainous"


# =========================
# NASA IMAGE FETCH
# =========================
def get_nasa_image(lat, lon):
    params = {
        "lat": lat,
        "lon": lon,
        "dim": 0.1,
        "api_key": NASA_API_KEY
    }

    try:
        r = requests.get(NASA_URL, params=params, timeout=10)
        if r.status_code == 200:
            return r.url
    except:
        pass

    return None


# =========================
# ELEVATION FETCH
# =========================
def get_elevation(lat, lon):
    try:
        r = requests.get(ELEVATION_URL, params={
            "locations": f"{lat},{lon}"
        }, timeout=10)

        data = r.json()
        return data["results"][0]["elevation"]
    except:
        return None


# =========================
# SCAN ROUTE (MAIN ENGINE)
# =========================
@app.route("/scan/<float:lat>/<float:lon>")
def scan(lat, lon):

    elevation = get_elevation(lat, lon)
    image_url = get_nasa_image(lat, lon)

    terrain = classify_terrain(elevation if elevation else 0)

    response = {
        "scan_time": datetime.utcnow().isoformat(),
        "coordinates": {
            "lat": lat,
            "lon": lon
        },
        "elevation_m": elevation,
        "terrain_type": terrain,
        "nasa_imagery": image_url,
        "scan_quality": "high" if elevation and image_url else "partial",
        "data_sources": ["NASA Earth API", "Open-Elevation"]
    }

    return jsonify(response)


# =========================
# RUN SERVER
# =========================
if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
